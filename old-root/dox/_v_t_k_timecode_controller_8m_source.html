129 //calculations always done in frames
130 -(NSNumber *)timecodeToFrames:(NSString *)tc withFrameRate: (NSNumber *)fr
131 {
132  //split string into tc elements
133  NSArray *split1 = [[NSArray alloc]initWithArray:[tc componentsSeparatedByString:@":"]];
134  NSArray *split2 = [[NSArray alloc]initWithArray:[[split1 lastObject] componentsSeparatedByString:@";"]];
135 
136  NSMutableArray *timeUnits = [[NSMutableArray alloc]init];
137  for (int i=0; i<[split1 count]-1; i++){
138  [timeUnits addObject:split1[i]];
139  }
140  [timeUnits addObjectsFromArray:split2];
141 
142  int frames = 3600*[fr intValue]*[timeUnits[0] intValue] +
143  60*[fr intValue]*[timeUnits[1] intValue] +
144  [fr intValue]*[timeUnits[2] intValue] +
145  [timeUnits[3] intValue];
146  return [NSNumber numberWithInt:frames];
147 }

149 //Convert frame to timecode
150 //calculations always done in frames, and converted back to formatted string for readability
151 -(NSString *)framesToTimecode:(NSNumber *)frames withFrameRate:(NSNumber *)fr
152 {
153  int tcHours, tcMinutes, tcSeconds, tcFrames, remainder;
154 
155  tcHours = [frames intValue]/(3600*[fr intValue]);
156  remainder = [frames intValue] - tcHours * (3600*[fr intValue]);
157 
158  tcMinutes = remainder / (60*[fr intValue]);
159  remainder = remainder - tcMinutes * (60*[fr intValue]);
160 
161  tcSeconds = remainder / [fr intValue];
162 
163  tcFrames = remainder - tcSeconds * [fr intValue];
164 
165  return [NSString stringWithFormat:@"%02d:%02d:%02d;%02d",tcHours,tcMinutes,tcSeconds,tcFrames];
166 }//framesToTimecode
